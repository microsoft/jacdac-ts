import { Link } from "gatsby"
import { graphql } from "gatsby"

export const query = graphql`
  query {
    allSpecJson {
      edges {
        node {
          shortId
          name
        }
      }
    }
  }
`

Known services are specified in [https://github.com/microsoft/jacdac/tree/main/services](https://github.com/microsoft/jacdac/tree/main/services)
using [this format](https://github.com/microsoft/jacdac/tree/main/spectool). 
Use the [Service Editor](/tools/service-editor) and send us a pull request 
to register your own service.

<ul>
    {props.data.allSpecJson.edges.map(node => <li>
        <Link key={node.node.shortId} to={`/services/${node.node.shortId}`}>{node.node.name}</Link>
    </li>)}
</ul>
